islamic_empire_asabiyya_level_calc = {
	## Putting variables/flags for asabiyya levels
	if = {
		limit = {
			var:islamic_empire_asabiyya <=0
		}
		remove_variable = islamic_empire_asabiyya
		remove_variable = asabiyya_level
		holder = {
			remove_character_modifier = islamic_empire_asabiyya_level_medium
			remove_character_modifier = islamic_empire_asabiyya_level_strong
			remove_character_modifier = islamic_empire_asabiyya_level_exceptional
			remove_character_modifier = islamic_empire_asabiyya_level_weak
			remove_character_modifier = islamic_empire_asabiyya_level_very_weak
			
			change_government = islamic_empire_government
			
		}
	}
	if = {
		limit = {
			var:islamic_empire_asabiyya < 25
		}
		set_variable = {
			name = asabiyya_level
			value = flag:asabiyya_level_very_weak_desc
		}
		holder = {
			remove_character_modifier = islamic_empire_asabiyya_level_medium
			remove_character_modifier = islamic_empire_asabiyya_level_strong
			remove_character_modifier = islamic_empire_asabiyya_level_exceptional
			remove_character_modifier = islamic_empire_asabiyya_level_weak
			add_character_modifier = { modifier = islamic_empire_asabiyya_level_very_weak }
			
		}
	}
	if = {
		limit = {
			var:islamic_empire_asabiyya >= 25
			var:islamic_empire_asabiyya < 50
		}
		set_variable = {
			name = asabiyya_level
			value = flag:asabiyya_level_weak_desc
		}
		holder = {
			remove_character_modifier = islamic_empire_asabiyya_level_medium
			remove_character_modifier = islamic_empire_asabiyya_level_strong
			remove_character_modifier = islamic_empire_asabiyya_level_exceptional
			remove_character_modifier = islamic_empire_asabiyya_level_very_weak
			add_character_modifier = { modifier = islamic_empire_asabiyya_level_weak }
		}
	}
	if = {
		limit = {
			var:islamic_empire_asabiyya >= 50
			var:islamic_empire_asabiyya < 75
		}
		set_variable = {
			name = asabiyya_level
			value = flag:asabiyya_level_medium_desc
		}
		holder = {
			remove_character_modifier = islamic_empire_asabiyya_level_very_weak
			remove_character_modifier = islamic_empire_asabiyya_level_strong
			remove_character_modifier = islamic_empire_asabiyya_level_exceptional
			remove_character_modifier = islamic_empire_asabiyya_level_weak
			add_character_modifier = { modifier = islamic_empire_asabiyya_level_medium }
		}
	}
	if = {
		limit = {
			var:islamic_empire_asabiyya >= 75
			var:islamic_empire_asabiyya < 150
		}
		set_variable = {
			name = asabiyya_level
			value = flag:asabiyya_level_strong_desc
		}
		holder = {
			remove_character_modifier = islamic_empire_asabiyya_level_medium
			remove_character_modifier = islamic_empire_asabiyya_level_very_weak
			remove_character_modifier = islamic_empire_asabiyya_level_exceptional
			remove_character_modifier = islamic_empire_asabiyya_level_weak
			add_character_modifier = { modifier = islamic_empire_asabiyya_level_strong }
		}
	}
	if = {
		limit = {
			var:islamic_empire_asabiyya >= 150
		}
		set_variable = {
			name = asabiyya_level
			value = flag:asabiyya_level_exceptional_desc
		}
		holder = {
			remove_character_modifier = islamic_empire_asabiyya_level_medium
			remove_character_modifier = islamic_empire_asabiyya_level_very_weak
			remove_character_modifier = islamic_empire_asabiyya_level_strong
			remove_character_modifier = islamic_empire_asabiyya_level_weak
			add_character_modifier = { modifier = islamic_empire_asabiyya_level_exceptional }
		}
	}
}

islamic_empire_calculate_regiment_cost = {
	save_scope_as = caliph

	set_variable = {
		name = islamic_empire_regiment_cost
		value = 0
	}

	every_vassal = {
		limit = {
			primary_title = {
				has_variable = islamic_empire_regiment_salary
			}
		}
		primary_title = {
			save_scope_as = regiment_title

			set_variable = {
				name = islamic_empire_regiment_salary_cost
				value = 0
			}
			change_variable = {
				name = islamic_empire_regiment_salary_cost
				add = var:islamic_empire_regiment_size_infantry
			}
			change_variable = {
				name = islamic_empire_regiment_salary_cost
				add = var:islamic_empire_regiment_size_light_cavalry
			}
			change_variable = {
				name = islamic_empire_regiment_salary_cost
				add = var:islamic_empire_regiment_size_heavy_cavalry
			}
			change_variable = {
				name = islamic_empire_regiment_salary_cost
				multiply = var:islamic_empire_regiment_salary
			}

			scope:caliph = {
				change_variable = {
					name = islamic_empire_regiment_cost
					add = prev.var:islamic_empire_regiment_salary_cost
				}
			}
		}
		
	}

	islamic_empire_remove_salary_cost = yes

	if = {			
		limit = {		
			var:islamic_empire_regiment_cost = 1	
		}		
		add_character_modifier = islamic_empire_salary_cost_modifier_1		
	}			
	if = {			
		limit = {		
			var:islamic_empire_regiment_cost = 2	
		}		
		add_character_modifier = islamic_empire_salary_cost_modifier_2
