ave_maria_alim_events.0002 = {
	type = character_event
	title = ave_maria_alim_events.0002.title
	desc = ave_maria_alim_events.0002.desc
	
	theme = realm
	
	left_portrait = {
		character = root
		animation = idle
	}
	right_portrait = {
		character = scope:target
		animation = idle
	}

	trigger = { exists = capital_province }

	immediate = {
		if = {
			limit = {
				is_landed = yes
				any_held_title = {
					tier = tier_county
					title_province = {
						NOR = {
							has_building_or_higher = madrasa_01
						}
					}
				}
			}
			# Create random ulama
			create_character = {
				employer = root
				template = islamic_empire_ulama_character_0
				save_scope_as = target
				gender_female_chance = 0
			}
		}
		if = {
			limit = {
				is_landed = yes
				any_held_title = {
					tier = tier_county
					title_province = {
						OR = {
							has_building = madrasa_01
							has_building = madrasa_02
						}
					}
				}
			}
			# Create random ulama
			create_character = {
				employer = root
				template = islamic_empire_ulama_character_1
				save_scope_as = target
				gender_female_chance = 0
			}
		}
		if = {
			limit = {
				is_landed = yes
				any_held_title = {
					tier = tier_county
					title_province = {
						OR = {
							has_building_or_higher = madrasa_03
							has_building = al_azhar_university
							has_building = house_of_wisdom_01
						}
					}
				}
			}
			# Create random ulama
			create_character = {
				employer = root
				template = islamic_empire_ulama_character
				save_scope_as = target
				gender_female_chance = 0
			}
		}
		if = {
			limit = {
				is_landed = no
			}
			# Create random ulama
			create_character = {
				employer = root
				template = islamic_empire_ulama_character_0
				save_scope_as = target
				gender_female_chance = 0
			}
		}
		scope:target = {
			add_random_hexaco_personality_trait = { WHO = this }
		}
	}

	option = {
		name = ave_maria_alim_events.0002.a
		remove_short_term_gold = medium_gold_value 
	}

	option = {
		name = ave_maria_alim_events.0002.b
		hidden_effect = {
			scope:target = {		
				death = {
					death_reason = death_disappearance
				}
			}
		}
	}
}
