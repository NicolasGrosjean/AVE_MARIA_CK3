monthly_pulse.12 = {
	type = empty
	hidden = yes
	
	immediate = {
		every_duchy = {
			limit = {
				is_landless_type_title = yes
				is_mercenary_company = no
				is_holy_order = no
				has_variable = faction
				holder = {
					has_trait = peasant_leader
					is_at_war = no
				}
			}
			holder = {
				death = natural
			}
		}
	}
	
	option = {
	}
